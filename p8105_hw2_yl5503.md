``` r
pols_df = 
  read_csv("./data/pols-month.csv") %>%
  janitor::clean_names()


pols_df <- pols_df %>% 
  separate(
    mon,
    into = c('year', 'month', 'day'), 
    sep = "-", 
    remove = TRUE) %>% 
  mutate(
    month = month.abb[as.numeric(month)], 
    president = case_when(prez_gop == 1 & prez_dem == 0 ~ "rep",
                          prez_dem == 1 & prez_gop == 0 ~ "dem")) %>%
    select(-prez_gop, -gov_gop, -day)
```
